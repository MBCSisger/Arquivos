create or replace TRIGGER NFECTL_T2 
AFTER INSERT ON NFECTL
FOR EACH ROW 
BEGIN
   IF INSERTING THEN
      INSERT INTO NFECAB
        ( 
         LJO,IDE_NNF,IDE_DEMI,CHAVE,IDE_MODELO,IDE_SERIE,PDV,IDE_CUF,IDE_cMunFG,emit_CNPJ,emit_IE,emit_xNome,emit_xFant,
         enderEmit_xLgr,enderEmit_nro,enderEmit_xBairro,enderEmit_cMun,enderEmit_xMun,enderEmit_UF,
         enderEmit_fone,emit_CRT,enderEmit_CEP , infcpl  
        )    
      VALUES 
        ( 
          (SELECT CODEMPRESA FROM LOGDAT WHERE CODIGO = 1),:NEW.NR_NFE,(SELECT DTMOV FROM LOGDAT WHERE CODIGO = 1),:NEW.CHAVE,
          :NEW.MODELO, :NEW.SERIE,:NEW.PDV,
          (SELECT COD_UF FROM CIDADE WHERE COD_IBGE = (SELECT  CID_IBGE FROM EMPRESA WHERE CODIGO = (SELECT CODEMPRESA FROM LOGDAT WHERE CODIGO = 1))),
          (SELECT CID_IBGE FROM EMPRESA WHERE CODIGO = (SELECT CODEMPRESA FROM LOGDAT WHERE CODIGO = 1)),
          (SELECT CGC FROM EMPRESA WHERE CODIGO = (SELECT CODEMPRESA FROM LOGDAT WHERE CODIGO = 1)),
          (SELECT IES FROM EMPRESA WHERE CODIGO = (SELECT CODEMPRESA FROM LOGDAT WHERE CODIGO = 1)),
          (SELECT NOME FROM EMPRESA WHERE CODIGO = (SELECT CODEMPRESA FROM LOGDAT WHERE CODIGO = 1)),
          (SELECT FANTASIA FROM EMPRESA WHERE CODIGO = (SELECT CODEMPRESA FROM LOGDAT WHERE CODIGO = 1)),
          (SELECT ENDERECO FROM EMPRESA WHERE CODIGO = (SELECT CODEMPRESA FROM LOGDAT WHERE CODIGO = 1)),
          (SELECT NUMERO FROM EMPRESA WHERE CODIGO = (SELECT CODEMPRESA FROM LOGDAT WHERE CODIGO = 1)),
          (SELECT BAIRRO FROM EMPRESA WHERE CODIGO = (SELECT CODEMPRESA FROM LOGDAT WHERE CODIGO = 1)),
          (SELECT CID_IBGE FROM EMPRESA WHERE CODIGO = (SELECT CODEMPRESA FROM LOGDAT WHERE CODIGO = 1)),
          (SELECT NOME FROM CIDADE WHERE COD_IBGE = (SELECT  CID_IBGE FROM EMPRESA WHERE CODIGO = (SELECT CODEMPRESA FROM LOGDAT WHERE CODIGO = 1))),
          (SELECT UF FROM CIDADE WHERE COD_IBGE = (SELECT  CID_IBGE FROM EMPRESA WHERE CODIGO = (SELECT CODEMPRESA FROM LOGDAT WHERE CODIGO = 1))),
          (SELECT SUBSTR(FONE,0,14) FROM EMPRESA WHERE CODIGO = (SELECT CODEMPRESA FROM LOGDAT WHERE CODIGO = 1)),
          (SELECT CRT FROM EMPRESA WHERE CODIGO = (SELECT CODEMPRESA FROM LOGDAT WHERE CODIGO = 1)),
          (SELECT CEP FROM EMPRESA WHERE CODIGO = (SELECT CODEMPRESA FROM LOGDAT WHERE CODIGO = 1)),
          (CASE WHEN (SELECT CRT FROM EMPRESA WHERE CODIGO = 1) = 1 THEN
           ('DOCUMENTO EMITIDO POR ME OU EPP OPTANTE PELO SIMPLES NACIONAL.')
          ELSE 
            ' '
          END)
        );
   END IF;
END NFECTL_T2;
