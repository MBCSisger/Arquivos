create or replace TRIGGER NFECTL_T3 
AFTER UPDATE 
OF CHV_NFE
ON NFECTL
FOR EACH ROW 
BEGIN
   -- TRIGGER PARA GRAVAR A CHAVE DA NFE NAS TABELAS
   IF UPDATING THEN
      UPDATE NFECAB SET
      CHV_NFE     = :NEW.CHV_NFE,
      IDE_DEMI    = (SELECT DTMOV FROM LOGDAT WHERE CODIGO = 1)
      WHERE CHAVE = :NEW.CHAVE;
      UPDATE NFEITE SET
      CHV_NFE     = :NEW.CHV_NFE
      WHERE CHAVE = :NEW.CHAVE;
      UPDATE NFEDUP SET
      CHV_NFE     = :NEW.CHV_NFE
      WHERE CHAVE = :NEW.CHAVE;
      UPDATE NFEREF SET
      CHV_NFE     = :NEW.CHV_NFE
      WHERE CHAVE = :NEW.CHAVE;
      UPDATE NFEVOL SET
      CHV_NFE     = :NEW.CHV_NFE
      WHERE CHAVE = :NEW.CHAVE;
   END IF;
END NFECTL_T3;
