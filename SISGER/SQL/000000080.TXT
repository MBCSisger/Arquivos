INSERT INTO CLIFONE
SELECT  CODIGO,
TRIM(SUBSTR(FONE,3,9)) AS FONE,
SUBSTR(FONE,1,2)  AS DDD,
(CASE WHEN SUBSTR(FONE,3,1) = '2' THEN 'FIX'
WHEN SUBSTR(FONE,3,1) = '3' THEN 'FIX' 
WHEN SUBSTR(FONE,3,1) = '4' THEN 'FIX'
WHEN SUBSTR(FONE,3,1) = '5' THEN 'FIX'
WHEN SUBSTR(FONE,3,1) = ' ' THEN 'FIX'
WHEN SUBSTR(FONE,3,1) = '6' THEN 'CEL'
WHEN SUBSTR(FONE,3,1) = '7' THEN 'CEL'
WHEN SUBSTR(FONE,3,1) = '8' THEN 'CEL'
WHEN SUBSTR(FONE,3,1) = '9' THEN 'CEL'
END) AS TIPO,0 AS SEQ
FROM CLIENTE
