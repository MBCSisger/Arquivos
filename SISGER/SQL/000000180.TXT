create or replace TRIGGER ORCAMENTO_T1 
AFTER INSERT OR DELETE ON ORCAMENTO
FOR EACH ROW 
BEGIN
   IF INSERTING THEN
      UPDATE ORCCTL SET
      DATA    = (SELECT DTMOV FROM LOGDAT WHERE CODIGO = 1),
      HORA    = (SELECT TO_CHAR(SYSDATE,'HH24:mi:ss') from dual),
      VRDESC  = VRDESC  + :NEW.VRDESC,
      VRTOTAL = VRTOTAL + :NEW.VRTOTAL
      WHERE CHAVE = :NEW.CHAVE;
   END IF;
   IF DELETING THEN 
      UPDATE ORCCTL SET
      VRDESC  = VRDESC  - :OLD.VRDESC,
      VRTOTAL = VRTOTAL - :OLD.VRTOTAL
      WHERE CHAVE = :OLD.CHAVE;
   END IF;
END ORCAMENTO_T1;
