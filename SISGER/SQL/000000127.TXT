CREATE OR REPLACE VIEW PRECO_ESPECIAL AS
SELECT CODCLI,CODPRO,DESCRICAO,VR_VENDA,EAN,VR,VR_ANT,DATA,
(CASE WHEN VR < VR_VENDA THEN 
   'V¡LIDO.'
ELSE
   'DESCARTADO.' 
END) AS STATUS  
FROM (
SELECT CODCLI,CODPRO,
(SELECT DESCRICAO FROM PRODUTO X WHERE X.CODIGO = A.CODPRO) AS DESCRICAO,
(SELECT VR_VENDA FROM PRODUTO X WHERE X.CODIGO = A.CODPRO) AS VR_VENDA,
EAN,VR,VR_ANT,DATA
FROM PRECOESP A
)
