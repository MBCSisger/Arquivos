create or replace TRIGGER NFECAB_T1 
AFTER UPDATE 
OF CHV_NFE
ON NFECAB
FOR EACH ROW 
BEGIN
   /* TRIGGER PARA GRAVAR DATA DE COMPRA DO CLIENTE */
   IF UPDATING THEN
      UPDATE CLIENTE SET
      DTCOMPRA = (SELECT DTMOV FROM LOGDAT WHERE CODIGO = 1)
      WHERE CODIGO = :NEW.CODIGO;
   END IF;
END NFECAB_T1;
