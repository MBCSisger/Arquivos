create or replace TRIGGER NFEITE_T2 
AFTER UPDATE 
OF CHV_NFE
ON NFEITE
FOR EACH ROW 
DECLARE
--1 = E_COMPRA -> ENTRADA POR COMPRA
--2 = E_TRANSF -> ENTRADA POR TRANSFERENCIA
--3 = E_DEVOL  -> ENTRADA POR DEVOLUÇĂO
--4 = E_BONI   -> ENTRADA POR BONIFICAÇĂO
--5 = S_DEVOL  -> SAIDA POR DEVOLUÇĂO
--6 = S_TRANSF -> SAIDA POR TRANSFERENCIA
--7 = S_VENDA  -> SAIDA POR VENDA

   X NUMBER;   
   CURSOR Y (Dn NUMBER) IS
--   SELECT COD_OPERACAO FROM NFECAB WHERE CHAVE = Dn;
--   SELECT COD_OPERACAO,IDE_TPNF FROM NFECAB WHERE CHAVE = Dn;
   SELECT TO_NUMBER(IDE_TPNF||COD_OPERACAO) FROM NFECAB WHERE CHAVE = Dn;
 BEGIN
   OPEN Y (:NEW.CHAVE); 
   FETCH Y INTO X;
   IF UPDATING THEN
      IF SUBSTR(X,1,1) = '0'  THEN -- ENTRADA
         IF SUBSTR(X,2,1) = '1'  THEN -- ENTRADA POR COMPRA
            UPDATE PRODUTO SET
            E_COMPRA = E_COMPRA + :NEW.QTRIB, 
            QT_SALDO = QT_SALDO + :NEW.QTRIB   
            WHERE CODIGO = :NEW.CPROD;
         END IF;
         IF SUBSTR(X,2,1) = '2'  THEN -- ENTRADA POR TRANSFERENCIA
            UPDATE PRODUTO SET
            E_TRANSF = E_TRANSF + :NEW.QTRIB, 
            QT_SALDO = QT_SALDO + :NEW.QTRIB   
           WHERE CODIGO = :NEW.CPROD;
        END IF;
        IF SUBSTR(X,2,1) = '3'  THEN -- ENTRADA POR DEVOLUCAO
           UPDATE PRODUTO SET
           E_DEVOL  = E_DEVOL  + :NEW.QTRIB, 
           QT_SALDO = QT_SALDO + :NEW.QTRIB   
           WHERE CODIGO = :NEW.CPROD;
        END IF;
        IF SUBSTR(X,2,1) = '4'  THEN -- ENTRADA POR BONIFICAÇĂO
           UPDATE PRODUTO SET
           E_BONI   = E_BONI   + :NEW.QTRIB, 
           QT_SALDO = QT_SALDO + :NEW.QTRIB   
           WHERE CODIGO = :NEW.CPROD;
           END IF;
      END IF;
      IF SUBSTR(X,1,1) = '1'  THEN -- SAIDA
         IF SUBSTR(X,2,1) = '5'  THEN -- SAIDA DE DEVOLUCAO DE MERCADORIA
            UPDATE PRODUTO SET
            S_DEVOL  = S_DEVOL  + :NEW.QTRIB, 
            QT_SALDO = QT_SALDO - :NEW.QTRIB   
            WHERE CODIGO = :NEW.CPROD;
         END IF;
         IF SUBSTR(X,2,1) = '6'  THEN -- SAIDA DE TRANSFERENCIA DE MERCADORIA ENTRE FILIAIS
            UPDATE PRODUTO SET
            S_TRANSF  = S_TRANSF + :NEW.QTRIB, 
            QT_SALDO = QT_SALDO  - :NEW.QTRIB   
            WHERE CODIGO = :NEW.CPROD;
         END IF;
         IF SUBSTR(X,2,1) = '7'  THEN -- VENDA DE MERCADORIA
            UPDATE PRODUTO SET
           S_VENDA  = S_VENDA  + :NEW.QTRIB, 
           QT_SALDO = QT_SALDO - :NEW.QTRIB,
           DT_VENDA = (SELECT DTMOV FROM LOGDAT WHERE CODIGO = 1)
           WHERE CODIGO = :NEW.CPROD;
         END IF;
      END IF;
   END IF;
   CLOSE Y;
END NFEITE_T2;
