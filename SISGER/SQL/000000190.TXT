CREATE VIEW PAGPDV
AS SELECT CHAVE,DATA,ITEM,PDV,MODELO,PAG_TPAG,PAG_VPAG,PAG_TPINTEGRA,PAG_CNPJ,PAG_TBAND,PAG_CAUT,PAG_VTROCO,PGTO,VRNF,(PGTO-VRNF) AS TROCO FROM (
SELECT CHAVE,(SELECT IDE_DEMI FROM NFECAB WHERE CHAVE = A.CHAVE)  AS DATA,
(SELECT MODELO FROM NFECTL WHERE CHAVE = A.CHAVE) AS MODELO,
ITEM,PDV,PAG_TPAG,PAG_VPAG,PAG_TPINTEGRA,PAG_CNPJ,PAG_TBAND,PAG_CAUT,PAG_VTROCO,
(SELECT SUM(PAG_VPAG)  FROM NFEPAG WHERE CHAVE = A.CHAVE) AS PGTO,
(SELECT TOTAL_VNF  FROM NFECAB WHERE CHAVE = A.CHAVE) AS VRNF
FROM NFEPAG A
WHERE 
(SELECT STATUS FROM NFECTL WHERE CHAVE = A.CHAVE) = 3 )
